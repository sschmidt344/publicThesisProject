Sub cleanup()'' test Macro''    Range("A1:A300").Select    Range("A300").Activate    ActiveSheet.Range("$A$1:$A$300").RemoveDuplicates Columns:=1, Header:=xlNo    Dim rng As Range    'Store blank cells inside a variable    On Error GoTo NoBlanksFound        Set rng = Range("A1:A300").SpecialCells(xlCellTypeBlanks)    On Error GoTo 0    'Delete blank cells and shift upward    rng.Rows.Delete Shift:=xlShiftUpExit Sub'ERROR HANLDERNoBlanksFound:  MsgBox "No Blank cells were found"End Sub
